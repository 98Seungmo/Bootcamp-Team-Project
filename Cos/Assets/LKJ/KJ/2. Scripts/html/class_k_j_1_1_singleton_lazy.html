초기값:
=
new Lazy<T>(() =>
{
T instance = FindObjectOfType(typeof(T)) as T;
if (instance == null)
{
GameObject obj = new GameObject("SingletonLazy");
instance = obj.AddComponent(typeof(T)) as T;
DontDestroyOnLoad(obj);
}
else
{
Destroy(instance as GameObject);
}
return instance;
})
