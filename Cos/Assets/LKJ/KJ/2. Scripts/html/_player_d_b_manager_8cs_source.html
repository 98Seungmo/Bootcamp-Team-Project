19 public static string GenerateShortUID(string longUID)
20 {
21 using (SHA256 sha256Hash = SHA256.Create())
22 {
23 /* SHA256 해시 값을 계산. */
24 byte[] bytes = sha256Hash.ComputeHash(Encoding.UTF8.GetBytes(longUID));
25
26 /* 바이트 배열을 String 으로 변환. */
27 StringBuilder builder = new StringBuilder();
28 for (int i = 0; i < bytes.Length; i++)
29 {
30 builder.Append(bytes[i].ToString("x2"));
31 }
32
33 /* 해시값의 앞부분만 사용하여 ShortUID 생성. */
34 return builder.ToString().Substring(0, 8); // 앞의 8자리만 사용
35 }
