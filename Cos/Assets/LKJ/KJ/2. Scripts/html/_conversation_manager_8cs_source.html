413 IEnumerator CheckClosestNpcRoutine()
414 {
415 while (true)
416 {
417 float minDistance = float.MaxValue;
418 Transform _closestNpc = null;
419 Vector3 playerForward = player.forward;
420
421 foreach (var npc in _npcs)
422 {
423 Vector3 directionToNpc = (npc.position - player.position).normalized;
424 float dotProduct = Vector3.Dot(playerForward, directionToNpc);
425
426 /* 플레이어가 대략적으로 바라보고, 상호작용 범위 내에 npc 가 있는지 */
427 if (dotProduct > 0.5f) // 1에 가까울 수록 정면이여야 함.
428 {
429 float distance = Vector3.Distance(player.position, npc.position);
430 if (distance < minDistance)
431 {
432 minDistance = distance;
433 _closestNpc = npc;
434 }
435 }
436 }
437
438 /* 가장 가까운 NPC가 일정 거리 이내에 있으면 UI 표시. */
439 if (_closestNpc != null && minDistance <= _interactDistance)
440 {
441 if (_currentInteractUI == null)
442 {
443 _currentInteractUI = Instantiate(interactUI);
444 /* canvas 부모로 설정. */
445 _currentInteractUI.transform.SetParent(canvas.transform, false);
446
447 }
448 /* NPC 의 스크린 좌표 위치로 UI를 이동 */
449 Vector2 screenPosition = Camera.main.WorldToScreenPoint(_closestNpc.position + Vector3.up * _offsetDistance);
450 Debug.Log($"스크린 좌표값 + {screenPosition}");
451 Vector2 canvasPosition;
452 RectTransform canvasRect = canvas.GetComponent<RectTransform>();
453 /* 캔버스 좌표계로 변환 */
454 RectTransformUtility.ScreenPointToLocalPointInRectangle(canvasRect, screenPosition, Camera.main, out canvasPosition);
455
456 /*캔버스 좌표계에서 추가적인 조정*/
457 canvasPosition += new Vector2(1000, 200);
458 Debug.Log($"캔버스 좌표값 {canvasPosition}");
459 /* RectTransfrom 의 anchoredPosition 사용, screenPosition 을 RectTranform 의 좌표계로 반환. */
460 /* UI 의 RectTransform 컴포넌트 가져옴. */
461 _currentInteractUI.GetComponent<RectTransform>().anchoredPosition = canvasPosition;
462 /* UI 활성화 */
463 Debug.Log("UI");
464 _currentInteractUI.SetActive(true);
465
466 }
467 else
468 {
469 if (_currentInteractUI != null)
470 {
471 Destroy(_currentInteractUI);
472 _currentInteractUI = null;
473 }
474 }
475
476 yield return new WaitForSeconds(checkInterval);
477 }
478 }
