16 public class SingletonLazy<T> : MonoBehaviour where T : class
17 {
18 /* Lazy<T> 를 사용하여 싱글톤 인스턴스를 저장.
19 Lazy 는 실제로 값이 필요할 때까지 생성을 지연시키는 객체 */
20 private static readonly Lazy<T> _instance =
21 new Lazy<T>(() =>
22 {
23 /* 현재 씬에서 T 타입의 객체를 찾음. */
24 T instance = FindObjectOfType(typeof(T)) as T;
25
26 /* 인스턴스가 null 이면 새로운 GameObject 를 생성하고 T 컴포넌트 추가 */
27 if (instance == null)
28 {
29 GameObject obj = new GameObject("SingletonLazy");
30 instance = obj.AddComponent(typeof(T)) as T;
31
32 /* 씬 전환시 Gameobject가 파괴되지 않도록 설정. */
33 DontDestroyOnLoad(obj);
34 }
35 else
36 {
37 /* 이미 인스턴스가 있을 경우 새로 생성된 GameObject 파괴 */
38 Destroy(instance as GameObject);
39 }
40
41 /* 생성된 인스턴스 반환. */
42 return instance;
43 });
44
48 public static T Instance
49 {
50 get

52 /* Lazy 인스턴스의 Value 속성을 통해 싱글톤 인스턴스 반환
