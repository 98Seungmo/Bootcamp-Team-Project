149 public void AddItem(string itemId)
150 {
151 Debug.Log($"{itemId} 추가");
152 /* 근데 해당 아이템이 뭔지 알아야함
153 플레이어 인벤토리에 해당 아이템이 있는지 체크해야 함
154 이미 있는 아이템이면 수량만 +1, 없으면 플레이어 인벤토리에 아이템이 추가됨.
155 새로운 아이템(즉 없는 아이템을 얻을 경우 슬롯도 같이 생성) */
156
157 Item itemToAdd = GetItItemById(itemId);
158
159 if (itemToAdd != null)
160 {
161 Item item = null;
162 foreach (var i in _inventory.items)
163 {
164 if (i.id == itemId)
165 {
166 item = i;
167 break;
168 }
169 }
170
171 if (item != null && item.id == itemId)
172 {
173 Debug.Log($"제발 :{item.id}");
174 Debug.Log($"제발2 :{item.imagePath}");
175
176 item.quantity++;
177 itemQuantity.text = item.quantity.ToString();
178 CreateSlot(item);
179 }
180 else
181 {
182
183 _inventory.items.Add(itemToAdd);
184
185 Debug.Log($"이미지 : {itemToAdd.id}");
186 CreateSlot(itemToAdd);
187 }
188
189 }
190 }

197 public void RemoveItem(string itemId)
198 {
199 Item itemToRemove = GetItItemById(itemId);
200
201 if (itemToRemove != null)
202 {
203 Item item = _inventory.items.Find(item => item.id == itemId);
204
205 if (item != null)
206 {
207 item.quantity--;
208 itemQuantity.text = item.quantity.ToString();
209
210 if (itemToRemove.quantity <= 0)
211 {
212 _inventory.items.Remove(itemToRemove);
213 RemoveSlot(itemToRemove);
214 }
215 }
216 else
217 {
218 Debug.Log("제거할 아이템 찾을 수 없음." + itemId);
219 }
220 }
221 }
